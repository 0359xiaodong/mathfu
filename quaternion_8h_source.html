39 /// @addtogroup mathfu_quaternion
40 /// @{
41 /// @class Quaternion
42 ///
43 /// @brief Stores a Quaternion of type T and provides a set of utility
44 /// operations on each Quaternion.
45 /// @tparam T Type of each element in the Quaternion.
46 template<class T>
47 class Quaternion {
48  public:
49  /// @brief Construct an uninitialized Quaternion.
50  inline Quaternion() {}
51 
52  /// @brief Construct a Quaternion from a copy.
53  /// @param q Quaternion to copy.
54  inline Quaternion(const Quaternion<T>& q) {
55  s_ = q.s_;
56  v_ = q.v_;
57  }
58 
59  /// @brief Construct a Quaternion using scalar values to initialize each
60  /// element.
61  ///
62  /// @param s1 Scalar component.
63  /// @param s2 First element of the Vector component.
64  /// @param s3 Second element of the Vector component.
65  /// @param s4 Third element of the Vector component.
66  inline Quaternion(const T& s1, const T& s2, const T& s3, const T& s4) {
67  s_ = s1;
68  v_ = Vector<T, 3>(s2, s3, s4);
69  }
70 
71  /// @brief Construct a quaternion from a scalar and 3-dimensional Vector.
72  ///
73  /// @param s1 Scalar component.
74  /// @param v1 Vector component.
75  inline Quaternion(const T& s1, const Vector<T, 3>& v1) {
76  s_ = s1;
77  v_ = v1;
78  }
79 
80  /// @brief Calculate the inverse Quaternion.
81  ///
82  /// This calculates the inverse such that <code>(q * q).Inverse()</code>
83  /// is the identity.
84  ///
85  /// @return Quaternion containing the result.
86  inline Quaternion<T> Inverse() const {
87  return Quaternion<T>(s_, -v_);
88  }
89 
90  /// @brief Multiply this Quaternion with another Quaternion.
91  ///
92  /// @note This is equivalent to
93  /// <code>FromMatrix(ToMatrix() * q.ToMatrix()).</code>
94  /// @param q Quaternion to multiply with.
95  /// @return Quaternion containing the result.
96  inline Quaternion<T> operator*(const Quaternion<T>& q) const {
97  return Quaternion<T>(
98  s_ * q.s_ - Vector<T, 3>::DotProduct(v_, q.v_),
99  s_ * q.v_ + q.s_ * v_ + Vector<T, 3>::CrossProduct(v_, q.v_));
100  }
101 
102  /// @brief Multiply this Quaternion by a scalar.
103  ///
104  /// This multiplies the angle of the rotation by a scalar factor.
105  /// @param s1 Scalar to multiply with.
106  /// @return Quaternion containing the result.
107  inline Quaternion<T> operator*(const T& s1) const {
108  T angle;
109  Vector<T, 3> axis;
110  ToAngleAxis(&angle, &axis);
111  angle *= s1;
112  return Quaternion<T>(cos(0.5f * angle),
113  axis.Normalized() * sin(0.5f * angle));
114  }
115 
116  /// @brief Multiply a Vector by this Quaternion.
117  ///
118  /// This will rotate the specified vector by the rotation specified by this
119  /// Quaternion.
120  ///
121  /// @param v1 Vector to multiply by this Quaternion.
122  /// @return Rotated Vector.
123  inline Vector<T, 3> operator*(const Vector<T, 3>& v1) const {
124  T ss = s_ + s_;
125  return ss * Vector<T, 3>::CrossProduct(v_, v1) + (ss * s_ - 1) * v1 +
126  2 * Vector<T, 3>::DotProduct(v_, v1) * v_;
127  }
128 
129  /// @brief Normalize this quaterion (in-place).
130  ///
131  /// @return Length of the quaternion.
132  inline T Normalize() {
133  T length = sqrt(s_ * s_ + v_ * v_);
134  T scale = (1 / length);
135  s_ *= scale;
136  v_ *= scale;
137  return length;
138  }
139 
140  /// @brief Convert this Quaternion to an Angle and axis.
141  ///
142  /// The returned angle is the size of the rotation in radians about the
143  /// axis represented by this Quaternion.
144  ///
145  /// @param angle Receives the angle.
146  /// @param axis Receives the normalized axis.
147  inline void ToAngleAxis(T* angle, Vector<T, 3>* axis) const {
148  *axis = s_ > 0 ? v_ : -v_;
149  *angle = 2 * atan2(axis->Normalize(), s_ > 0 ? s_ : -s_);
150  }
151 
152  /// @brief Convert this Quaternion to 3 Euler Angles.
153  ///
154  /// @return 3-dimensional Vector where each element is a angle of rotation
155  /// (in radians) around the x, y, and z axes.
156  inline Vector<T, 3> ToEulerAngles() const {
157  Matrix<T, 3> m(ToMatrix());
158  T cos2 = m[0] * m[0] + m[1] * m[1];
159  if (cos2 < 1e-6f) {
160  return Vector<T, 3>(
161  0, m[2] < 0 ? static_cast<T>(0.5 * M_PI) :
162  static_cast<T>(-0.5 * M_PI), -atan2(m[3], m[4]));
163  } else {
164  return Vector<T, 3>(
165  atan2(m[5], m[8]), atan2(-m[2], sqrt(cos2)), atan2(m[1], m[0]));
166  }
167  }
168 
169  /// @brief Convert to a 3x3 Matrix.
170  ///
171  /// @return 3x3 rotation Matrix.
172  inline Matrix<T, 3> ToMatrix() const {
173  const T x2 = v_[0] * v_[0], y2 = v_[1] * v_[1], z2 = v_[2] * v_[2];
174  const T sx = s_ * v_[0], sy = s_ * v_[1], sz = s_ * v_[2];
175  const T xz = v_[0] * v_[2], yz = v_[1] * v_[2], xy = v_[0] * v_[1];
176  return Matrix<T, 3>(1 - 2 * (y2 + z2), 2 * (xy + sz), 2 * (xz - sy),
177  2 * (xy - sz), 1 - 2 * (x2 + z2), 2 * (sx + yz),
178  2 * (sy + xz), 2 * (yz - sx), 1 - 2*(x2 + y2));
179  }
180 
181  /// @brief Create a Quaternion from an angle and axis.
182  ///
183  /// @param angle Angle in radians to rotate by.
184  /// @param axis Axis in 3D space to rotate around.
185  /// @return Quaternion containing the result.
186  static Quaternion<T> FromAngleAxis(const T& angle,
187  const Vector<T, 3>& axis) {
188  const T halfAngle = static_cast<T>(0.5) * angle;
189  Vector<T, 3> localAxis(axis);
190  return Quaternion<T>(cos(halfAngle),
191  localAxis.Normalized() * sin(halfAngle));
192  }
193 
194  /// @brief Create a quaternion from 3 euler angles.
195  ///
196  /// @param angles 3-dimensional Vector where each element contains an
197  /// angle in radius to rotate by about the x, y and z axes.
198  /// @return Quaternion containing the result.
199  static Quaternion<T> FromEulerAngles(const Vector<T, 3>& angles) {
200  const Vector<T, 3> halfAngles(static_cast<T>(0.5) * angles[0],
201  static_cast<T>(0.5) * angles[1],
202  static_cast<T>(0.5) * angles[2]);
203  const T sinx = sin(halfAngles[0]), cosx = cos(halfAngles[0]);
204  const T siny = sin(halfAngles[1]), cosy = cos(halfAngles[1]);
205  const T sinz = sin(halfAngles[2]), cosz = cos(halfAngles[2]);
206  return Quaternion<T>(
207  cosx * cosy * cosz + sinx * siny * sinz,
208  sinx * cosy * cosz - cosx * siny * sinz,
209  cosx * siny * cosz + sinx * cosy * sinz,
210  cosx * cosy * sinz - sinx * siny * cosz);
211  }

213  /// @brief Create a quaternion from a rotation Matrix.
214  ///
215  /// @param m 3x3 rotation Matrix.
216  /// @return Quaternion containing the result.
217  static Quaternion<T> FromMatrix(const Matrix<T, 3>& m) {
218  const T trace = m(0, 0) + m(1, 1) + m(2, 2);
219  if (trace > 0) {
220  const T s = sqrt(trace + 1) * 2;
221  const T oneOverS = 1 / s;
222  return Quaternion<T>(
223  static_cast<T>(0.25) * s, (m[5] - m[7]) * oneOverS,
224  (m[6] - m[2]) * oneOverS, (m[1] - m[3]) * oneOverS);
225  } else if (m[0] > m[4] && m[0] > m[8]) {
226  const T s = sqrt(m[0] - m[4] - m[8] + 1) * 2;
227  const T oneOverS = 1 / s;
228  return Quaternion<T>(
229  (m[5] - m[7]) * oneOverS, static_cast<T>(0.25) * s,
230  (m[3] + m[1]) * oneOverS, (m[6] + m[2]) * oneOverS);
231  } else if (m[4] > m[8]) {
232  const T s = sqrt(m[4] - m[0] - m[8] + 1) * 2;
233  const T oneOverS = 1 / s;
234  return Quaternion<T>(
235  (m[6] - m[2]) * oneOverS, (m[3] + m[1]) * oneOverS,
236  static_cast<T>(0.25) * s, (m[5] + m[7]) * oneOverS);
237  } else {
238  const T s = sqrt(m[8] - m[0] - m[4] + 1) * 2;
239  const T oneOverS = 1 / s;
240  return Quaternion<T>(
241  (m[1] - m[3]) * oneOverS, (m[6] + m[2]) * oneOverS,
242  (m[5] + m[7]) * oneOverS, static_cast<T>(0.25) * s);
243  }
244  }

246  /// @brief Calculate the spherical linear interpolation between two
247  /// Quaternions.
248  ///
249  /// @param q1 Start Quaternion.
250  /// @param q2 End Quaternion.
251  /// @param s1 The scalar value determining how far from q1 and q2 the
252  /// resulting quaternion should be. A value of 0 corresponds to q1 and a
253  /// value of 1 corresponds to q2.
254  /// @result Quaternion containing the result.
255  static inline Quaternion<T> Slerp(
256  const Quaternion<T>& q1, const Quaternion<T>& q2, const T& s1) {
257  if(q1.s_ * q2.s_ + Vector<T, 3>::DotProduct(q1.v_, q2.v_) > 0.999999f)
258  return Quaternion<T>(
259  q1.s_ * (1 - s1) + q2.s_ * s1, q1.v_ * (1 - s1) + q2.v_ * s1);
260  return q1 * ((q1.Inverse() * q2) * s1);
261  }
262 
263  /// @brief Access an element of the quaternion.
264  ///
265  /// @param i Index of the element to access.
266  /// @return A reference to the accessed data that can be modified by the
267  /// caller.
268  inline T& operator[](const int i) {
269  if (i == 0) return s_;
270  return v_[i - 1];
271  }
272 
273  /// @brief Access an element of the quaternion.
274  ///
275  /// @param i Index of the element to access.
276  /// @return A const reference to the accessed.
277  inline const T& operator[](const int i) const {
278  return const_cast<Quaternion<T>*>(this)->operator[](i);
279  }
